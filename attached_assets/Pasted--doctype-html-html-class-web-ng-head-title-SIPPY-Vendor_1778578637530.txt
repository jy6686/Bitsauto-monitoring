    function menu_search(event) {
        var len = event.target.value.length;

        var e = document.getElementById("menu-search");

        var ul = document.getElementById('menu-search-results');
        if (ul) {
            e.removeChild(ul);
        }

        if (len == 0) {
            document.getElementById("menu-search-icon").classList.remove('fa-times-circle');
            document.getElementById("menu-search-icon").classList.add('fas-search');
            document.getElementById("menu-search-icon").removeEventListener('click', menu_search);
            return;
        } else if (len < 3) {
            document.getElementById("menu-search-icon").classList.remove('fas-search');
            document.getElementById("menu-search-icon").classList.add('fa-times-circle');
            document.getElementById("menu-search-icon").addEventListener('click', function() {
                document.getElementById("menu-search-pattern").value = '';
                var event = new Event('input');
                document.getElementById("menu-search-pattern").dispatchEvent(event);
            });
            return;
        } else {
            document.getElementById("menu-search-icon").classList.remove('fas-search');
            document.getElementById("menu-search-icon").classList.add('fa-times-circle');
            document.getElementById("menu-search-icon").addEventListener('click', function() {
                document.getElementById("menu-search-pattern").value = '';
                var event = new Event('input');
                document.getElementById("menu-search-pattern").dispatchEvent(event);
            });

            ul = document.createElement("ul");
            ul.setAttribute('id', 'menu-search-results');
            ul.classList.add("wide");

            var m, i;
            try {
                m = document.getElementById("cssmenu").getElementsByTagName('li');
                for (i = 0; i < m.length; i++) {
                    if (m[i].classList.contains("has-sub")) {
                        continue;
                    };

                    var n, j;
                    n = m[i].getElementsByTagName('a');
                    for (j = 0; j < n.length; j++) {
                        if (!n[j].hasAttribute('href')) {
                            continue;
                        }
                        var re = new RegExp(event.target.value, 'i');
                        if (n[j].innerText.match(re)) {
                            ul.appendChild(m[i].cloneNode(true));
                        }
                    }
                }
            } finally {
                if (ul.childNodes.length > 0) {
                    e.appendChild(ul);
                }
            }
        }
    }

    $(document).ready(function() {
