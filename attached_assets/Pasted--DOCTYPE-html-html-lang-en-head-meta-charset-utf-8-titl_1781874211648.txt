        function status_modal(i){
             var elem = $('#id_company_status_modal').bodal({
                blurring: true,
                onShow: function () {
                    $.ajax({
                        method:'get',
                        dataType:'json',
                        url:'/switch_api/'+i+'/',
                        success:function(data){
                            var reseller = false;
                            var client = false;
                            var st_red = '<i class="icon-circle" style="background:red"></i>';
                            var st_green = '<i class="icon-circle" style="background:green"></i>';
                            var client_tbody = $('#id_company_status_modal .client-status table tbody').html('');
                            var reseller_tbody = $('#id_company_status_modal .reseller-status table tbody').html('');

                            responsiveTable(elem);

                            $.each(data,function(ind,val){
                                var status = val.error.length==0?(val.status?st_green:st_red):val.error;
                                if(val.is_reseller){
                                    reseller = true;
                                    $(reseller_tbody).append('<tr><td data-title="SBC">'+val.name+'</td><td data-title="status">'+status+'</td></tr>')
                                    $('.reseller-status').removeClass('hide');
                                }
                                else{
                                    client = true;
                                    $(client_tbody).append('<tr><td data-title="SBC">'+val.name+'</td><td data-title="status">'+status+'</td></tr>')
                                    $('.client-status').removeClass('hide');
                                }
                            });
                            if(data.length==0){
                                $('#id_company_status_modal .alert').removeClass('hide');
                            }
                            else{
                                $('#id_company_status_modal .alert').addClass('hide');
                            }
                            $('#id_company_status_modal .loader').addClass('hide');
                            $('#id_company_status_modal .ui.dimmer').addClass('hide');
                        }
                    })

                }
            })
            .bodal('setting', 'closable', false)
            .bodal('show');
        };

            var bindSortselect= function(){
                var selectedOption;
                $("select").each(function(){
                    selectedOption = $(this).children("option").filter(function(index,option){
                        return $(option).prop("selected");
                    });

                    $(this).html($(this).children("option").sort(function (a, b) {
                        if (a.text.toUpperCase() == b.text.toUpperCase()) return 0;
                        if (a.text.toUpperCase() < b.text.toUpperCase()) return -1;
                        if (a.text.toUpperCase() > b.text.toUpperCase()) return 1;
                    }));
                    //Set selected option
                    $(selectedOption).prop("selected", "selected");
                });
            };
