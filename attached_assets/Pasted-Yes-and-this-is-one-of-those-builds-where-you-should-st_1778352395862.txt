🔁 STEP 2 — Convert existing SipMessage[] → Ladder Events

Add this function (this is the core logic):

function buildLadder(messages: SipMessage[]): SipLadderEvent[] {
  const events: SipLadderEvent[] = [];

  let prevTs = messages[0]?.timestamp ?? 0;

  for (const msg of messages) {
    const deltaMs = msg.timestamp - prevTs;
    prevTs = msg.timestamp;

    const status =
      msg.code >= 500 ? "error" :
      msg.code >= 400 ? "error" :
      msg.code >= 300 ? "warning" :
      "success";

    // simple routing inference
    const from: SipLane =
      msg.direction === "in" ? "carrier" : "caller";

    const to: SipLane =
      msg.method === "INVITE" || msg.method === "ACK"
        ? "sippy"
        : msg.direction === "out"
        ? "carrier"
        : "caller";

    events.push({
      id: msg.id,
      from,
      to,
      method: msg.method,
      code: msg.code,
      label: `${msg.method}${msg.code ? ` ${msg.code}` : ""}`,
      timestamp: msg.timestamp,
      deltaMs,
      status,
    });
  }

  return events;
}
