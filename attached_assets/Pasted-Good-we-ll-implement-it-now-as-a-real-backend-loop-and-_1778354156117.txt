🏗️ 1. Create the engine
server/synthetic/scheduler.ts
import { db } from "../db";
import { testCampaigns, testCampaignResults } from "@shared/schema";
import { runTestCall } from "../test-calls/run-test-call"; // existing engine
import { sql } from "drizzle-orm";

let isRunning = false;

export async function runSyntheticScheduler() {
  if (isRunning) return;
  isRunning = true;

  try {
    const campaigns = await db.select().from(testCampaigns);

    for (const campaign of campaigns) {
      if (!campaign.isActive) continue;

      const start = Date.now();

      try {
        // 1. Execute synthetic call (already exists in your system)
        const result = await runTestCall({
          routeId: campaign.routeId,
          destination: campaign.destination,
          carrierId: campaign.carrierId,
        });

        const duration = Date.now() - start;

        // 2. Normalize metrics
        const asr = result.answered ? 1 : 0;
        const pdd = result.pdd || null;
        const mos = result.mos || null;

        // 3. Store result
        await db.insert(testCampaignResults).values({
          campaignId: campaign.id,
          routeId: campaign.routeId,
          carrierId: campaign.carrierId,
          asr,
          pdd,
          mos,
          durationMs: duration,
          status: result.status,
          raw: result,
          createdAt: new Date(),
        });

        // 4. Optional: lightweight degradation signal
        if (asr === 0 || (pdd && pdd > 5000)) {
          console.log("[synthetic] degradation detected", campaign.id);
        }
      } catch (err) {
        console.error("[synthetic] failed campaign", campaign.id, err);

        await db.insert(testCampaignResults).values({
          campaignId: campaign.id,
          routeId: campaign.routeId,
          carrierId: campaign.carrierId,
          asr: 0,
          pdd: null,
          mos: null,
          status: "failed",
          raw: String(err),
          createdAt: new Date(),
        });
      }
    }
  } finally {
    isRunning = false;
  }
}
⏱️ 2. Add scheduler (cron loop)
server/synthetic/runner.ts
import { runSyntheticScheduler } from "./scheduler";

export function startSyntheticEngine() {
  console.log("[synthetic] engine started");

  // initial delay (avoid startup spike)
  setTimeout(() => {
    runSyntheticScheduler();
  }, 10_000);

  // every 5 minutes
  setInterval(() => {
    runSyntheticScheduler();
  }, 5 * 60 * 1000);
}
