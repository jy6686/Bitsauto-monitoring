Logic (simple but powerful)
export async function runCorrelationEngine(db: any) {
  const windowMinutes = 30;

  const since = new Date(Date.now() - windowMinutes * 60000);

  const signals = await db.query.ai_ops_events.findMany({
    where: (t, { gte }) => gte(t.created_at, since),
  });

  const anomalies = await db.query.anomaly_events.findMany({
    where: (t, { gte }) => gte(t.created_at, since),
  });

  // GROUPING KEY: entity OR vendor OR operationType fallback
  const groups: Record<string, any> = {};

  function getKey(item: any) {
    return item.entity || item.vendor || item.type || "unknown";
  }

  // 1. add signals
  for (const s of signals) {
    const key = getKey(s);

    if (!groups[key]) {
      groups[key] = {
        entity: key,
        signals: [],
        anomalies: [],
      };
    }

    groups[key].signals.push(s);
  }

  // 2. add anomalies
  for (const a of anomalies) {
    const key = getKey(a);

    if (!groups[key]) {
      groups[key] = {
        entity: key,
        signals: [],
        anomalies: [],
      };
    }

    groups[key].anomalies.push(a);
  }

  // 3. build incidents
  for (const key of Object.keys(groups)) {
    const g = groups[key];

    if (g.signals.length === 0 && g.anomalies.length === 0) continue;

    const severity =
      g.signals.some((s: any) => s.severity === "high") ||
      g.anomalies.some((a: any) => a.severity === "critical")
        ? "high"
        : "medium";

    const title =
      `Incident: ${key} degradation`;

    await db.insert(ai_ops_incidents).values({
      title,
      entity: key,
      severity,
      start_time: new Date(),
      last_seen: new Date(),

      signals_count: g.signals.length,
      anomalies_count: g.anomalies.length,

      status: "active",
    });
  }
}

app.get("/api/aiops/incidents", async (req, res) => {
  const incidents = await db
    .select()
    .from(ai_ops_incidents)
    .orderBy(desc(ai_ops_incidents.last_seen));

  res.json(incidents);
});
