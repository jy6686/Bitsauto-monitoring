    function menu_search_reset() {
        var e = document.getElementById("menu_search");

        var ul = document.getElementById('menu_search_results');
        if (ul) {
            e.removeChild(ul);
        }
    }

    function menu_search(event) {
        var len = event.target.value.length;

        var e = document.getElementById("menu_search");

        var ul = document.getElementById('menu_search_results');
        if (ul) {
            e.removeChild(ul);
        }

        if (len == 0) {
            document.getElementById("menu_search_icon").classList.remove('fa-times-circle');
            document.getElementById("menu_search_icon").classList.add('fas-search');
            document.getElementById("menu_search_icon").removeEventListener('click');
            return;
        } else if (len < 3) {
            document.getElementById("menu_search_icon").classList.remove('fas-search');
            document.getElementById("menu_search_icon").classList.add('fa-times-circle');
            document.getElementById("menu_search_icon").addEventListener('click', function() {
                document.getElementById("menu_search_pattern").value = '';
                var event = new Event('input');
                document.getElementById("menu_search_pattern").dispatchEvent(event);
            });
            return;
        } else {
            document.getElementById("menu_search_icon").classList.remove('fas-search');
            document.getElementById("menu_search_icon").classList.add('fa-times-circle');
            document.getElementById("menu_search_icon").addEventListener('click', function() {
                document.getElementById("menu_search_pattern").value = '';
                var event = new Event('input');
                document.getElementById("menu_search_pattern").dispatchEvent(event);
            });

            ul = document.createElement("ul");
            ul.setAttribute('id', 'menu_search_results');
            ul.classList.add("wide");

            var m, i;
            try {
                m = document.getElementById("cssmenu").getElementsByTagName('li');
                for (i = 0; i < m.length; i++) {
                    if (m[i].classList.contains("has-sub")) {
                        continue;
                    };

                    var n, j;
                    n = m[i].getElementsByTagName('a');
                    for (j = 0; j < n.length; j++) {
                        if (!n[j].hasAttribute('href')) {
                            continue;
                        }
                        var re = new RegExp(event.target.value, 'i');
                        if (n[j].innerText.match(re)) {
                            ul.appendChild(m[i].cloneNode(true));
                        }
                    }
                }
            } finally {
                if (ul.childNodes.length > 0) {
                    e.appendChild(ul);
                }
            }
        }
    }

    addLoadEvent(function() {
        var e = document.getElementById('menu_search_pattern');
        e.addEventListener('input', menu_search);
    });
</script>

function enableAttrs(prefix, id) {
    var divs = document.getElementsByTagName('div');
    var re = new RegExp(prefix + "[^\\s]+");
    for (var i = 0; i < divs.length; i++) {
        var cl = divs[i].className;
        if (cl.length > 0 && cl.search(re) >= 0) {
            cl = new String(cl.match(re));
            var id2 = cl.substr(prefix.length)
            if (id2 == id) {
                divs[i].style.display = 'inline';
            } else {
                divs[i].style.display = 'none';
            }
        }
    }
}

function initOnnetRouting(onLoad) {
    var form=document.forms['rg_edit'];

    var onnet_i_connection = form.onnet_i_connection;
    var onnet_voicemail_i_connection = form.onnet_voicemail_i_connection;
    var onnet_scope = form.onnet_scope;
    var onnet_timeout_100 = form.onnet_timeout_100;
    var onnet_timeout_1xx = form.onnet_timeout_1xx;
    var onnet_timeout_2xx = form.onnet_timeout_2xx;

    if (onnet_i_connection.value == '') {
        onnet_voicemail_i_connection.disabled = true;
        onnet_scope.disabled = true;
        onnet_timeout_100.disabled = true;
        onnet_timeout_1xx.disabled = true;
        onnet_timeout_2xx.disabled = true;
        return;
    }

    onnet_voicemail_i_connection.disabled = false;
    onnet_scope.disabled = false;
    onnet_timeout_100.disabled = false;
    onnet_timeout_1xx.disabled = false;
    onnet_timeout_2xx.disabled = false;
    if (!onLoad && onnet_i_connection.value == '0' && onnet_voicemail_i_connection.value == '') {
        onnet_voicemail_i_connection.value = '0';
    }
}

function setConnectionsList(id) {
    var form=document.forms['rg_edit'];

    removeAllOptions(form.rt_entr_conn);
    for (i = 0; i < conns.length; i++) {
        if (conns[i][0] == id) {
            addOption(form.rt_entr_conn, conns[i][2], conns[i][1]);
        }
    }

    d = document.getElementById("rt_action");
    if (id == -1) {
        form.rt_entr_conn.disabled = true;
        d.style.visibility = "hidden";
    } else {
        form.rt_entr_conn.disabled = false;
        d.style.visibility = "visible";
    }
}

function submitRGForm(action, id) {
    var form=document.forms['rg_edit'];

    form.action.value=action;

    if (id) {
        form.i_routing_group_member.value=id;
    }
    form.keep_rg_history.value=1;

    // fill in policy
    var o = [];
    var obj = form.policy1;
    if (hasOptions(obj)) {
        for (var i=0; i<obj.options.length; i++) {
            o[i] = obj.options[i].value;
        }
        form.policy.value = o.join(',');
    };

    form.submit();
}

}
