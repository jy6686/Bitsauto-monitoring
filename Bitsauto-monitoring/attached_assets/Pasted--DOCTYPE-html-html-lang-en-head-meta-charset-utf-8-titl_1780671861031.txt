        function responsiveTable(elem) {
            $(elem).find(".ba-table:not(.no-responsive) tbody tr").each(function () {
                var $firstTd = $(this).find("td:first-child:not(.dataTables_empty)")
                var $secondTd = $(this).find("td:eq(1)");

                var trButtonLenght = $(this).find('span.tr-button').length;

                if(!trButtonLenght) {
                    if($firstTd.attr('data-responsive') == "hide"){
                        $secondTd.append("<span class='tr-button'></span>")
                    }
                    else {
                        $firstTd.append("<span class='tr-button'></span>");

                }

        // Check Required Fields Function --- START
        function checkRequiredFields($selector) {
            $selector.find('.help-block').remove();
            $selector.find('.field.error').removeClass( 'error' );
            $selector.find('.field.required').not('.not-validate').each( function() {
                var $this = $(this),
                    thisVal = $this.find('input, select, textarea').val();
                if( !thisVal ) {
                    $this.addClass( 'error' );
                    $this.append( '<p class="f-right help-block">This field is required.</p>' );
                }
            });
            if ( $selector.find('.field.error').length == 0 ) {
                return true;
            }
            else {
                return false;
            }

            var bindSortselect= function(){
                var selectedOption;
                $('select:not([data-sorting="false"])').each(function(){
                    selectedOption = $(this).children("option").filter(function(index,option){
                        return $(option).prop("selected");
                    });

                    $(this).html($(this).children("option").sort(function (a, b) {
                        if (a.text.toUpperCase() == b.text.toUpperCase()) return 0;
                        if (a.text.toUpperCase() < b.text.toUpperCase()) return -1;
                        if (a.text.toUpperCase() > b.text.toUpperCase()) return 1;
                    }));
                    //Set selected option
                    $(selectedOption).prop("selected", "selected");
                });
            };
