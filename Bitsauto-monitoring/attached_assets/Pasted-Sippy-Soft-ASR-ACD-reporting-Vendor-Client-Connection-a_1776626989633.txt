Margin % = (Margin / Revenue) * 100
Backend calculation service:
export function calculateMargin(data) {
    const revenue = data.reduce((a, b) => a + b.revenue, 0);
    const cost = data.reduce((a, b) => a + b.cost, 0);

    const margin = revenue - cost;

    return {
        revenue,
        cost,
        margin,
        margin_percent: revenue ? (margin / revenue) * 100 : 0
    };
}

📡 SIPPY SERVICE (CORE INTEGRATION)
import axios from "axios";

export async function fetchSippyCDR(params) {
    const response = await axios.post(process.env.SIPPY_API_URL, {
        username: process.env.SIPPY_USER,
        password: process.env.SIPPY_PASS,
        ...params
    });

    return response.data.cdrs;
}

export function calculateMetrics(cdrs) {
    return cdrs.map(cdr => ({
        vendor: cdr.vendor,
        connection: cdr.connection,
        calls: cdr.calls,
        billable: cdr.billable,
        duration: cdr.duration,
        acd: cdr.duration / (cdr.answered || 1),
        asr: (cdr.answered / cdr.calls) * 100,
        pdd: cdr.pdd,
        revenue: cdr.revenue,
        cost: cdr.cost
    }));
}
