⚙️ STEP 2 — Signal Mapper (pure logic)

export function mapExecToSignal(execResult: any, operationType: string) {
  const signals = [];

  // 1. Failure signal
  if (!execResult.success) {
    signals.push({
      type: "ROUTING_FAILURE",
      severity: "high",
      message: execResult.message || "Execution failed",
      entity: operationType,
      value: null
    });
  }

  // 2. Latency signal
  if (execResult.durationMs && execResult.durationMs > 6000) {
    signals.push({
      type: "EXECUTION_LATENCY_HIGH",
      severity: "medium",
      message: `Slow execution: ${execResult.durationMs}ms`,
      entity: operationType,
      value: String(execResult.durationMs)
    });
  }

  // 3. Vendor/routing degradation hint
  if (
    !execResult.success &&
    operationType?.includes("routing_group_member")
  ) {
    signals.push({
      type: "VENDOR_DEGRADATION_SIGNAL",
      severity: "high",
      message: "Routing operation failed",
      entity: operationType,
      value: null
    });
  }

  return signals;
}
